import { createLocalVue, mount } from '@vue/test-utils'
import Profile from './profile.component.html'

describe('Profile Component', () => {
  let wrapper

  beforeEach(() => {
    wrapper = mount(Profile, {
      localVue: createLocalVue(),
      propsData: {
        role: 'editor'
      }
    })
  })

  afterEach(() => {
    wrapper.destroy()
  })

  it('renders email input when role is editor', () => {
    expect(wrapper.find('input[type="email"]').exists()).toBe(true)
  })

  it('does not render email input when role is not editor', () => {
    wrapper.setProps({ role: 'user' })
    expect(wrapper.find('input[type="email"]').exists()).toBe(false)
  })

  it('calls updateProfile function when form is submitted', () => {
    const updateProfileSpy = jest.spyOn(wrapper.vm, 'updateProfile')
    wrapper.find('form').simulate('submit')
    expect(updateProfileSpy).toHaveBeenCalledTimes(1)
  })
})
